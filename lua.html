الدرس الأول: بيئة التنفيذ

-- طريقة تنفيذ الكود بأمان
local function RunCode(source)
    local success, result = pcall(function()
        return loadstring(source)()
    end)
    return success and result or "خطأ في التنفيذ"
end

الدرس الثالث: التلاعب بالأحداث البعيدة

-- التعامل مع Remote Events
local ReplicatedStorage = game:GetService("ReplicatedStorage")
local Remote = ReplicatedStorage:WaitForChild("DamageEvent")

-- تتبع الاستدعاءات
local oldNamecall
oldNamecall = hookmetamethod(game, "__namecall", function(self, ...)
    local method = getnamecallmethod()
    if method == "FireServer" and self == Remote then
        print("تم اعتراض استدعاء!")
        return oldNamecall(self, ...)
    end
    return oldNamecall(self, ...)
end)
